class scoreboard;

mailbox mon2scb;

function new(mailbox mon2scb);
this.mon2scb = mon2scb;
endfunction

task main();
transaction trans;
repeat (1) begin
mon2scb.get(trans);

if ( ((trans.a ^ trans.b) == trans.sum) &&
((trans.a & trans.b) == trans.carry) )
$display("RESULT : ACCEPTED");
else
$display("RESULT : WRONG");

trans.display("Scoreboard");
end
endtask

endclass
